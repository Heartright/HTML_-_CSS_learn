1. Примешивание .red должно скомпилироваться в color: red. $red: red; p{ color: $red } 2. Примешивание .blue должно скомпилироваться в color: blue. $blue: blue; p{ color: $blue } 3. Примешивание .green должно скомпилироваться в color: green. $green: green; p{ color: $green } 4. Функция .h(число) должна компилироваться в height: число. Пример: .h(200px) скомпилируется в height: 200px. @mixin h($h){ height: $h; } p{ @include h(200px); } 5. Функция .w(число) должна компилироваться в width: число. Пример: .w(200px) скомпилируется в width: 200px. @mixin w($w){ width: $w; } p{ @include w(200px); } 6. Функция .placeholder() должна компилироваться в кроссбраузерный вариант псевдоэлемента placeholder. @mixin placeholder($color,$bg:red){ &::-moz-placeholder{ color: #{$color}; background: #{$bg}; }; &::-webkit-placeholder{ color: #{$color}; background: #{$bg}; }; &::placeholder{ color: #{$color}; background: #{$bg}; }; } input{ @include placeholder(blue); } 7. Функция .selection() должна компилироваться в кроссбраузерный вариант псевдоэлемента selection. @mixin selection ($color: black, $w: 100px){ &::-moz-selection{ color: #{$color}; width: #{$w}; } &::-webkit-selection{ color: #{$color}; width: #{$w}; } &::selection{ color: #{$color}; width: #{$w};} } p { @include selection (white, 200px); } 8. Функция .keyframes() должна компилироваться в кроссбраузерный вариант команды keyframes. @mixin keyframes($test) { @-webkit-keyframes #{$test} { @content; } @-moz-keyframes #{$test} { @content; } @-ms-keyframes #{$test} { @content; } @keyframes #{$test} { @content; } } @include keyframes(move) { 0% { transform: rotate(0deg); } 100% { transform: rotate(180deg); } } @for $i from 1 through 100 { .f#{$i} {font-size: 1px * $i; } }
